print('——————————————————————————————————————————密码加/解密程序———————————————————————————————————————————')

choose=input('请输入您想要使用的密码功能(加/解)：')


if choose=='加':
    previouspassword=input('请输入密码(纯数字且首位不为0)：')
    key=input('请输入密钥(同上)：')
    password=int(previouspassword)^int(key)
    print('加密后密码是:'+str(password))


if choose=='解':
    pass_word=input('请输入被加密的密码：')
    key_word=input('请输入密钥：')
    previous_password=int(pass_word)^int(key_word)
    print('解密后的密码是：'+str(previous_password))
